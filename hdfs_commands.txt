-- Create HDFS folders
hdfs dfs -mkdir -p /project/ecommerce/input
hdfs dfs -mkdir -p /project/ecommerce/backup
hdfs dfs -mkdir -p /project/ecommerce/output

-- Upload file to HDFS
hdfs dfs -put /tmp/orders.csv /project/ecommerce/input/

-- Check file in HDFS
hdfs dfs -ls /project/ecommerce/input
hdfs dfs -cat /project/ecommerce/input/orders.csv
-- Make backup
hdfs dfs -cp /project/ecommerce/input/orders.csv /project/ecommerce/backup/

-- Rename backup file
hdfs dfs -mv /project/ecommerce/backup/orders.csv /project/ecommerce/backup/orders_backup.csv

-- Check folders
hdfs dfs -ls -R /project/ecommerce

-- Check size
hdfs dfs -du -h /project/ecommerce
